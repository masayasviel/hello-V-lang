## fizzbuzz

[公式](https://github.com/vlang/v/blob/master/examples/fizz_buzz.v)

```v
fn main() {
	mut s := ''
	for n in 1 .. 101 {
		if n % 3 == 0 {
			s += 'Fizz'
		}
		if n % 5 == 0 {
			s += 'Buzz'
		}
		if s == '' {
			println(n)
		}
		else {
			println(s)
		}
		s = ''
	}
}
```

自己流

```v
fn fizzbuzz(i int) string {
	mut s := ""
	if i % 3 == 0 {
		s += "fizz"
	}
	if i % 5 == 0 {
		s += "buzz"
	}
	if s == "" {
		s = i.str()
	}
	return s
}

fn main() {
	for i := 1; i <= 10; i++ {
		println(fizzbuzz(i))
	}
}
```
